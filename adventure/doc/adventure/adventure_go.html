    package cyoa

    import (

    "encoding/json"
"html/template"
"io"
"net/http"

    type Story mapChapter

    type Chapter struct {

    Title      string   `json:"title"`
Paragraphs []string `json:"story"`
Options    []Option `json:"options"`

    }

    type Option struct {

    Text    string `json:"text"`
Chapter string `json:"chapter"`

    }

    var defaultHandlerTemplate = ` <!DOCTYPE html> <html lang=“en”> <head>

    <meta charset="UTF-8">
<title>Choose Your Own Adventure</title>

    </head> <body>

    <h1>{{.Title}}</h1>
{{range .Paragraphs}}
<p>{{.}}</p>
{{end}}
<ul>
{{range .Options}}
<li><a href="/{{.Chapter}}">{{.Text}}</a></li>
{{end}}
</ul>

    </body> </html> `

    type handler struct {

    s Story

    }

    func NewHandler(s Story) http.Handler {

    return handler{s}

    }

    func (h handler) ServeHTTP(w http.ResponseWriter, r *http.Request) {

    tpl := template.Must(template.New("").Parse(defaultHandlerTemplate))
err := tpl.Execute(w, h.s["intro"])
if err != nil {
        panic(err)
}

    }

    // Create Story structure from JSON definition func JSONStory(r io.Reader) (Story, error) {

    var story Story
decoder := json.NewDecoder(r)
if err := decoder.Decode(&story); err != nil {
        return nil, err
}
return story, nil

    }
